76      @ParameterizedTest(name = "{0} ({1}) -> {2}")
77      @ArgumentsSource(TestCaseArgumentsProvider.class)
78      void assertSupportedSQL(final String sqlCaseId, final SQLCaseType sqlCaseType, final String databaseType) {
79          if (IGNORE_TEST_CASES.contains(sqlCaseId)) {
80              return;
81          }
82          String sql = SQL_CASES.getSQL(sqlCaseId, sqlCaseType, SQL_PARSER_TEST_CASES.get(sqlCaseId).getParameters());
83          SQLStatement actual = parseSQLStatement("H2".equals(databaseType) ? "MySQL" : databaseType, sql);
84          SQLParserTestCase expected = SQL_PARSER_TEST_CASES.get(sqlCaseId);
85          SQLStatementAssert.assertIs(new SQLCaseAssertContext(sqlCaseId, sql, expected.getParameters(), sqlCaseType), actual, expected);
86      }
87      
88      private SQLStatement parseSQLStatement(final String databaseType, final String sql) {
89          return "ShardingSphere".equals(databaseType)
90                  ? new DistSQLStatementParserEngine().parse(sql)
91                  : new SQLStatementVisitorEngine(databaseType).visit(new SQLParserEngine(databaseType, new CacheOption(128, 1024L)).parse(sql, false));
92      }
93      
94      private static class TestCaseArgumentsProvider implements ArgumentsProvider {
95          
96          @Override
97          public Stream<? extends Arguments> provideArguments(final ExtensionContext extensionContext) {
98              InternalSQLParserITSettings settings = extensionContext.getRequiredTestClass().getAnnotation(InternalSQLParserITSettings.class);
99              Preconditions.checkNotNull(settings, "Annotation InternalSQLParserITSettings is required.");
100             return getTestParameters(settings.value()).stream();
101         }
102         
103         private Collection<Arguments> getTestParameters(final String... databaseTypes) {
104             Collection<Arguments> result = new LinkedList<>();
105             for (InternalSQLParserTestParameter each : SQL_CASES.generateTestParameters(Arrays.stream(databaseTypes).collect(Collectors.toSet()))) {
106                 if (!isPlaceholderWithoutParameter(each)) {
107                     result.add(Arguments.arguments(each.getSqlCaseId(), each.getSqlCaseType(), each.getDatabaseType()));
108                 }
109             }
110             return result;
111         }
112         
113         private boolean isPlaceholderWithoutParameter(final InternalSQLParserTestParameter testParam) {
114             return SQLCaseType.PLACEHOLDER == testParam.getSqlCaseType() && SQL_PARSER_TEST_CASES.get(testParam.getSqlCaseId()).getParameters().isEmpty();
115         }
