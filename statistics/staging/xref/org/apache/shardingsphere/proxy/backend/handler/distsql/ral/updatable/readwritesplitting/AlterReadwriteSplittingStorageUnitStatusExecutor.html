53  /**
54   * Alter readwrite-splitting storage unit status executor.
55   */
56  // TODO move to readwritesplitting module
57  @DistSQLExecutorClusterModeRequired
58  @Setter
59  public final class AlterReadwriteSplittingStorageUnitStatusExecutor implements DistSQLUpdateExecutor<AlterReadwriteSplittingStorageUnitStatusStatement>, DistSQLExecutorDatabaseAware {
60      
61      private static final String DISABLE = "DISABLE";
62      
63      private ShardingSphereDatabase database;
64      
65      @Override
66      public void checkBeforeUpdate(final AlterReadwriteSplittingStorageUnitStatusStatement sqlStatement, final ContextManager contextManager) {
67          checkReadwriteSplittingRule();
68      }
69      
70      private void checkReadwriteSplittingRule() {
71          Optional<ReadwriteSplittingRule> rule = database.getRuleMetaData().findSingleRule(ReadwriteSplittingRule.class);
72          ShardingSpherePreconditions.checkState(rule.isPresent(), () -> new UnsupportedSQLOperationException("The current database has no read_write splitting rules"));
73      }
74      
75      @Override
76      public void executeUpdate(final AlterReadwriteSplittingStorageUnitStatusStatement sqlStatement, final ContextManager contextManager) {
77          String toBeUpdatedStorageUnit = sqlStatement.getStorageUnitName();
78          Map<String, String> replicaStorageUnits = getReplicaResources();
79          Map<String, String> disabledStorageUnits = getDisabledResources(contextManager);
80          Map<String, String> autoAwareResources = getAutoAwareResources();
81          boolean isDisable = DISABLE.equals(sqlStatement.getStatus());
82          if (isDisable) {
83              checkDisable(contextManager, database.getName(), disabledStorageUnits.keySet(), toBeUpdatedStorageUnit, replicaStorageUnits);
84          } else {
85              checkEnable(contextManager, database.getName(), disabledStorageUnits, toBeUpdatedStorageUnit);
86          }
87          Collection<String> groupNames = getGroupNames(toBeUpdatedStorageUnit, replicaStorageUnits, disabledStorageUnits, autoAwareResources);
88          String groupName = sqlStatement.getGroupName();
89          if (Strings.isNullOrEmpty(groupName)) {
90              updateStatus(contextManager, database.getName(), groupNames, toBeUpdatedStorageUnit, isDisable);
91          } else {
92              checkGroupName(groupNames, groupName);
93              updateStatus(contextManager, database.getName(), Collections.singleton(groupName), toBeUpdatedStorageUnit, isDisable);
94          }
95      }
96      
97      private Map<String, String> getReplicaResources() {
98          Map<String, String> result = new HashMap<>();
99          getExportedReadwriteSplittingRules().entrySet().stream().filter(entry -> !entry.getValue().isEmpty()).forEach(entry -> addReplicaResource(result, entry));
100         return result;
101     }
102     
103     private Map<String, Map<String, String>> getExportedReadwriteSplittingRules() {
104         Map<String, Map<String, String>> result = new HashMap<>();
105         database.getRuleMetaData().findSingleRule(ReadwriteSplittingRule.class)
106                 .filter(each -> new RuleExportEngine(each)
107                         .containExportableKey(Arrays.asList(ExportableConstants.EXPORT_DYNAMIC_READWRITE_SPLITTING_RULE, ExportableConstants.EXPORT_STATIC_READWRITE_SPLITTING_RULE)))
108                 .map(each -> new RuleExportEngine(each).export(Arrays.asList(ExportableConstants.EXPORT_DYNAMIC_READWRITE_SPLITTING_RULE, ExportableConstants.EXPORT_STATIC_READWRITE_SPLITTING_RULE)))
109                 .ifPresent(optional -> {
110                     result.putAll((Map) optional.getOrDefault(ExportableConstants.EXPORT_DYNAMIC_READWRITE_SPLITTING_RULE, Collections.emptyMap()));
111                     result.putAll((Map) optional.getOrDefault(ExportableConstants.EXPORT_STATIC_READWRITE_SPLITTING_RULE, Collections.emptyMap()));
112                 });
113         return result;
114     }
115     
116     private Map<String, String> getDisabledResources(final ContextManager contextManager) {
117         return getDisabledStorageNodes(contextManager.getMetaDataContexts().getPersistService()).stream()
118                 .collect(Collectors.toMap(QualifiedDatabase::getDataSourceName, QualifiedDatabase::getGroupName, (value1, value2) -> String.join(",", value1, value2)));
119     }
120     
121     private Map<String, String> getAutoAwareResources() {
122         Map<String, String> result = new HashMap<>();
123         getExportedReadwriteSplittingRules().values().stream().filter(each -> each.containsKey(ExportableItemConstants.AUTO_AWARE_DATA_SOURCE_NAME)).forEach(each -> Splitter.on(",")
124                 .splitToList(each.get(ExportableItemConstants.REPLICA_DATA_SOURCE_NAMES)).forEach(each1 -> put(result, each1, each.get(ExportableItemConstants.AUTO_AWARE_DATA_SOURCE_NAME))));
125         return result;
126     }
127     
128     private void checkGroupName(final Collection<String> groupNames, final String groupName) {
129         ShardingSpherePreconditions.checkState(groupNames.contains(groupName),
130                 () -> new UnsupportedSQLOperationException(String.format("The current database does not exist the group `%s`", groupName)));
131     }
132     
133     private void checkEnable(final ContextManager contextManager, final String databaseName, final Map<String, String> disabledResources, final String toBeEnabledResource) {
134         checkResourceExists(contextManager, databaseName, toBeEnabledResource);
135         checkIsNotDisabled(disabledResources.keySet(), toBeEnabledResource);
136     }
137     
138     private void checkIsNotDisabled(final Collection<String> disabledResources, final String toBeEnabledResource) {
139         ShardingSpherePreconditions.checkState(disabledResources.contains(toBeEnabledResource), () -> new UnsupportedSQLOperationException(String.format("`%s` is not disabled", toBeEnabledResource)));
140     }
141     
142     private void checkResourceExists(final ContextManager contextManager, final String databaseName, final String toBeDisabledResource) {
143         Collection<String> notExistedResources = contextManager
144                 .getMetaDataContexts().getMetaData().getDatabase(databaseName).getResourceMetaData().getNotExistedDataSources(Collections.singleton(toBeDisabledResource));
145         ShardingSpherePreconditions.checkState(notExistedResources.isEmpty(), () -> new MissingRequiredStorageUnitsException(databaseName, Collections.singleton(toBeDisabledResource)));
146     }
147     
148     private void checkDisable(final ContextManager contextManager, final String databaseName, final Collection<String> disabledStorageUnits, final String toBeDisabledStorageUnit,
149                               final Map<String, String> replicaResources) {
150         checkResourceExists(contextManager, databaseName, toBeDisabledStorageUnit);
151         checkIsDisabled(replicaResources, disabledStorageUnits, toBeDisabledStorageUnit);
152         checkIsReplicaResource(replicaResources, toBeDisabledStorageUnit);
153     }
154     
155     private void checkIsDisabled(final Map<String, String> replicaResources, final Collection<String> disabledStorageUnits, final String toBeDisabledStorageUnit) {
156         String toBeDisableResourceRuleNames = replicaResources.get(toBeDisabledStorageUnit);
157         ShardingSpherePreconditions.checkState(!Strings.isNullOrEmpty(toBeDisableResourceRuleNames) || !disabledStorageUnits.contains(toBeDisabledStorageUnit),
158                 () -> new UnsupportedSQLOperationException(String.format("`%s` has been disabled", toBeDisabledStorageUnit)));
159     }
160     
161     private void checkIsReplicaResource(final Map<String, String> replicaStorageUnits, final String toBeDisabledStorageUnit) {
162         ShardingSpherePreconditions.checkState(replicaStorageUnits.containsKey(toBeDisabledStorageUnit),
163                 () -> new UnsupportedSQLOperationException(String.format("`%s` is not used as a read storage unit by any read-write separation rules,cannot be disabled", toBeDisabledStorageUnit)));
164     }
165     
166     private Collection<String> getGroupNames(final String toBeDisableStorageUnit, final Map<String, String> replicaStorageUnits,
167                                              final Map<String, String> disabledStorageUnits, final Map<String, String> autoAwareResources) {
168         String groupNames = autoAwareResources.getOrDefault(toBeDisableStorageUnit, replicaStorageUnits.getOrDefault(toBeDisableStorageUnit, disabledStorageUnits.get(toBeDisableStorageUnit)));
169         return Splitter.on(",").splitToList(groupNames);
170     }
171     
172     private void updateStatus(final ContextManager contextManager, final String databaseName, final Collection<String> groupNames, final String toBeDisableStorageUnit, final boolean isDisable) {
173         groupNames.forEach(each -> {
174             StorageNodeDataSource storageNodeDataSource = new StorageNodeDataSource(StorageNodeRole.MEMBER, isDisable ? DataSourceState.DISABLED : DataSourceState.ENABLED);
175             contextManager.getInstanceContext().getEventBusContext()
176                     .post(new DataSourceDisabledEvent(databaseName, each, toBeDisableStorageUnit, storageNodeDataSource));
177         });
178     }
179     
180     private Collection<QualifiedDatabase> getDisabledStorageNodes(final MetaDataBasedPersistService persistService) {
181         Map<String, StorageNodeDataSource> storageNodes = new StorageNodeStatusService(persistService.getRepository()).loadStorageNodes();
182         return storageNodes.entrySet().stream().filter(each -> DataSourceState.DISABLED == each.getValue().getStatus())
183                 .map(each -> new QualifiedDatabase(each.getKey())).filter(each -> database.getName().equalsIgnoreCase(each.getDatabaseName())).collect(Collectors.toList());
184     }
185     
186     private void addReplicaResource(final Map<String, String> replicaStorageUnits, final Entry<String, Map<String, String>> readwriteSplittingRule) {
187         readwriteSplittingRule.getValue().entrySet().stream().filter(entry -> ExportableItemConstants.REPLICA_DATA_SOURCE_NAMES.equals(entry.getKey()))
188                 .map(entry -> Arrays.asList(entry.getValue().split(","))).flatMap(Collection::stream).forEach(each -> put(replicaStorageUnits, each, readwriteSplittingRule.getKey()));
189     }
190     
191     private void put(final Map<String, String> map, final String key, final String value) {
192         map.put(key, map.containsKey(key) ? String.join(",", map.get(key), value) : value);
193     }
194     
195     @Override
196     public Class<AlterReadwriteSplittingStorageUnitStatusStatement> getType() {
197         return AlterReadwriteSplittingStorageUnitStatusStatement.class;
198     }
