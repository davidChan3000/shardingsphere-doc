42  /**
43   * Column segment binder.
44   */
45  @NoArgsConstructor(access = AccessLevel.PRIVATE)
46  public final class ColumnSegmentBinder {
47      
48      private static final Collection<String> EXCLUDE_BIND_COLUMNS = new LinkedHashSet<>(Arrays.asList("ROWNUM", "ROW_NUMBER", "ROWNUM_"));
49      
50      /**
51       * Bind column segment with metadata.
52       *
53       * @param segment table segment
54       * @param statementBinderContext statement binder context
55       * @param tableBinderContexts table binder contexts
56       * @param outerTableBinderContexts outer table binder contexts
57       * @return bounded column segment
58       */
59      public static ColumnSegment bind(final ColumnSegment segment, final SQLStatementBinderContext statementBinderContext,
60                                       final Map<String, TableSegmentBinderContext> tableBinderContexts, final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
61          if (EXCLUDE_BIND_COLUMNS.contains(segment.getIdentifier().getValue().toUpperCase())) {
62              return segment;
63          }
64          ColumnSegment result = new ColumnSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getIdentifier());
65          segment.getOwner().ifPresent(result::setOwner);
66          Collection<TableSegmentBinderContext> tableBinderContextValues = getTableSegmentBinderContexts(segment, statementBinderContext, tableBinderContexts, outerTableBinderContexts);
67          Optional<ColumnSegment> inputColumnSegment = findInputColumnSegment(segment.getIdentifier().getValue(), tableBinderContextValues);
68          result.setColumnBoundedInfo(createColumnSegmentBoundedInfo(segment, inputColumnSegment.orElse(null)));
69          return result;
70      }
71      
72      private static Collection<TableSegmentBinderContext> getTableSegmentBinderContexts(final ColumnSegment segment, final SQLStatementBinderContext statementBinderContext,
73                                                                                         final Map<String, TableSegmentBinderContext> tableBinderContexts,
74                                                                                         final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
75          if (segment.getOwner().isPresent()) {
76              return getTableBinderContextByOwner(segment.getOwner().get().getIdentifier().getValue(), tableBinderContexts, outerTableBinderContexts);
77          }
78          if (!statementBinderContext.getJoinTableProjectionSegments().isEmpty()) {
79              return Collections.singleton(new TableSegmentBinderContext(statementBinderContext.getJoinTableProjectionSegments()));
80          }
81          return tableBinderContexts.values();
82      }
83      
84      private static Collection<TableSegmentBinderContext> getTableBinderContextByOwner(final String owner, final Map<String, TableSegmentBinderContext> tableBinderContexts,
85                                                                                        final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
86          if (tableBinderContexts.containsKey(owner)) {
87              return Collections.singleton(tableBinderContexts.get(owner));
88          }
89          if (outerTableBinderContexts.containsKey(owner)) {
90              return Collections.singleton(outerTableBinderContexts.get(owner));
91          }
92          return Collections.emptyList();
93      }
94      
95      private static Optional<ColumnSegment> findInputColumnSegment(final String columnName, final Collection<TableSegmentBinderContext> tableBinderContexts) {
96          ProjectionSegment projectionSegment = null;
97          ColumnSegment result = null;
98          for (TableSegmentBinderContext each : tableBinderContexts) {
99              projectionSegment = each.getProjectionSegmentByColumnLabel(columnName);
100             if (projectionSegment instanceof ColumnProjectionSegment) {
101                 ShardingSpherePreconditions.checkState(null == result, () -> new AmbiguousColumnException(columnName));
102                 result = ((ColumnProjectionSegment) projectionSegment).getColumn();
103             }
104         }
105         // TODO optimize exception message according to different segment
106         ShardingSpherePreconditions.checkState(null != projectionSegment, () -> new UnknownColumnException(columnName));
107         return Optional.ofNullable(result);
108     }
109     
110     private static ColumnSegmentBoundedInfo createColumnSegmentBoundedInfo(final ColumnSegment segment, final ColumnSegment inputColumnSegment) {
111         IdentifierValue originalDatabase = null == inputColumnSegment ? null : inputColumnSegment.getColumnBoundedInfo().getOriginalDatabase();
112         IdentifierValue originalSchema = null == inputColumnSegment ? null : inputColumnSegment.getColumnBoundedInfo().getOriginalSchema();
113         IdentifierValue originalTable =
114                 null == segment.getColumnBoundedInfo().getOriginalTable() ? Optional.ofNullable(inputColumnSegment).map(optional -> optional.getColumnBoundedInfo().getOriginalTable()).orElse(null)
115                         : segment.getColumnBoundedInfo().getOriginalTable();
116         IdentifierValue originalColumn =
117                 null == segment.getColumnBoundedInfo().getOriginalColumn() ? Optional.ofNullable(inputColumnSegment).map(optional -> optional.getColumnBoundedInfo().getOriginalColumn()).orElse(null)
118                         : segment.getColumnBoundedInfo().getOriginalColumn();
119         return new ColumnSegmentBoundedInfo(originalDatabase, originalSchema, originalTable, originalColumn);
120     }
121     
122     /**
123      * Bind using column segment with metadata.
124      *
125      * @param segment using column segment
126      * @param tableBinderContexts table binder contexts
127      * @return bounded using column segment
128      */
129     public static ColumnSegment bindUsingColumn(final ColumnSegment segment, final Map<String, TableSegmentBinderContext> tableBinderContexts) {
130         ColumnSegment result = new ColumnSegment(segment.getStartIndex(), segment.getStopIndex(), segment.getIdentifier());
131         segment.getOwner().ifPresent(result::setOwner);
132         Collection<TableSegmentBinderContext> tableBinderContextValues = tableBinderContexts.values();
133         Collection<ColumnSegment> usingInputColumnSegments = findUsingInputColumnSegments(segment.getIdentifier().getValue(), tableBinderContextValues);
134         if (usingInputColumnSegments.size() >= 2) {
135             Iterator<ColumnSegment> iterator = usingInputColumnSegments.iterator();
136             result.setColumnBoundedInfo(createColumnSegmentBoundedInfo(segment, iterator.next()));
137             result.setOtherUsingColumnBoundedInfo(createColumnSegmentBoundedInfo(segment, iterator.next()));
138         }
139         return result;
140     }
141     
142     private static Collection<ColumnSegment> findUsingInputColumnSegments(final String columnName, final Collection<TableSegmentBinderContext> tableBinderContexts) {
143         ProjectionSegment projectionSegment;
144         Collection<ColumnSegment> result = new LinkedList<>();
145         for (TableSegmentBinderContext each : tableBinderContexts) {
146             projectionSegment = each.getProjectionSegmentByColumnLabel(columnName);
147             if (projectionSegment instanceof ColumnProjectionSegment) {
148                 result.add(((ColumnProjectionSegment) projectionSegment).getColumn());
149             }
150         }
151         return result;
152     }
