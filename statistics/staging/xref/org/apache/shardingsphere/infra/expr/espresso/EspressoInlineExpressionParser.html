18  package org.apache.shardingsphere.infra.expr.espresso;
19  
20  import com.google.common.base.Strings;
21  import com.google.common.collect.Sets;
22  import org.apache.shardingsphere.infra.exception.core.ShardingSpherePreconditions;
23  import org.apache.shardingsphere.infra.expr.spi.InlineExpressionParser;
24  import org.apache.shardingsphere.infra.util.groovy.GroovyUtils;
25  
26  import java.net.URL;
27  import java.util.ArrayList;
28  import java.util.Collection;
29  import java.util.Collections;
30  import java.util.LinkedHashSet;
31  import java.util.List;
32  import java.util.Properties;
33  import java.util.Set;
34  import java.util.stream.Collectors;
35  
36  /**
37   * Espresso inline expression parser.
38   */
39  public final class EspressoInlineExpressionParser implements InlineExpressionParser {
40      
41      private static final String JAVA_CLASSPATH;
42      
43      private String inlineExpression;
44      
45      static {
46          URL groovyJarUrl = EspressoInlineExpressionParser.class.getClassLoader().getResource("build/libs/groovy.jar");
47          ShardingSpherePreconditions.checkNotNull(groovyJarUrl, NullPointerException::new);
48          JAVA_CLASSPATH = groovyJarUrl.getPath();
49      }
50      
51      @Override
52      public void init(final Properties props) {
53          inlineExpression = props.getProperty(INLINE_EXPRESSION_KEY);
54      }
55      
56      @Override
57      public String handlePlaceHolder() {
58          return handlePlaceHolder(inlineExpression);
59      }
60      
61      /**
62       * Replace all inline expression placeholders.
63       *
64       * @param inlineExpression inline expression with {@code $->}
65       * @return result inline expression with {@code $}
66       */
67      private String handlePlaceHolder(final String inlineExpression) {
68          return inlineExpression.contains("$->{") ? inlineExpression.replaceAll("\\$->\\{", "\\$\\{") : inlineExpression;
69      }
70      
71      @Override
72      public List<String> splitAndEvaluate() {
73          try (ReflectContext context = new ReflectContext(JAVA_CLASSPATH)) {
74              if (Strings.isNullOrEmpty(inlineExpression)) {
75                  return Collections.emptyList();
76              }
77              return flatten(evaluate(context, GroovyUtils.split(handlePlaceHolder(inlineExpression))));
78          }
79      }
80      
81      private List<ReflectValue> evaluate(final ReflectContext context, final List<String> inlineExpressions) {
82          List<ReflectValue> result = new ArrayList<>(inlineExpressions.size());
83          for (String each : inlineExpressions) {
84              StringBuilder expression = new StringBuilder(handlePlaceHolder(each));
85              if (!each.startsWith("\"")) {
86                  expression.insert(0, '"');
87              }
88              if (!each.endsWith("\"")) {
89                  expression.append('"');
90              }
91              result.add(evaluate(context, expression.toString()));
92          }
93          return result;
94      }
95      
96      private ReflectValue evaluate(final ReflectContext context, final String expression) {
97          return context.getBindings("java")
98                  .getMember("groovy.lang.GroovyShell")
99                  .newInstance()
100                 .invokeMember("parse/(Ljava/lang/String;)Lgroovy/lang/Script;", expression)
101                 .invokeMember("run/()Ljava/lang/Object;");
102     }
103     
104     /**
105      * Flatten.
106      *
107      * @param segments Actually corresponds to some class instance of {@link java.lang.Object}.
108      *                 This Object may or may not correspond to a class instance of `groovy.lang.GString`.
109      * @return List of String
110      */
111     private List<String> flatten(final List<ReflectValue> segments) {
112         List<String> result = new ArrayList<>();
113         for (ReflectValue each : segments) {
114             if (!each.isString()) {
115                 result.addAll(assemblyCartesianSegments(each));
116             } else {
117                 result.add(each.as(String.class));
118             }
119         }
120         return result;
121     }
122     
123     /**
124      * Assembly cartesian segments.
125      *
126      * @param segment Actually corresponds to a class instance of `groovy.lang.GString`.
127      * @return List of String
128      */
129     private List<String> assemblyCartesianSegments(final ReflectValue segment) {
130         Set<List<String>> cartesianValues = getCartesianValues(segment);
131         List<String> result = new ArrayList<>(cartesianValues.size());
132         for (List<String> each : cartesianValues) {
133             result.add(assemblySegment(each, segment));
134         }
135         return result;
136     }
137     
138     /**
139      * Get cartesian values.
140      *
141      * @param segment Actually corresponds to a class instance of `groovy.lang.GString`.
142      * @return A Set consisting of a List of Strings
143      */
144     @SuppressWarnings("unchecked")
145     private Set<List<String>> getCartesianValues(final ReflectValue segment) {
146         Object[] segmentAsObjectArray = segment.invokeMember("getValues/()[Ljava/lang/Object;").as(Object[].class);
147         List<Set<String>> result = new ArrayList<>(segmentAsObjectArray.length);
148         for (Object each : segmentAsObjectArray) {
149             if (null == each) {
150                 continue;
151             }
152             if (each instanceof Collection) {
153                 result.add(((Collection<Object>) each).stream().map(Object::toString).collect(Collectors.toCollection(LinkedHashSet::new)));
154             } else {
155                 result.add(Sets.newHashSet(each.toString()));
156             }
157         }
158         return Sets.cartesianProduct(result);
159     }
160     
161     /**
162      * Assembly segment.
163      *
164      * @param cartesianValue List of String
165      * @param segment        Actually corresponds to a class instance of `groovy.lang.GString`.
166      * @return {@link java.lang.String}
167      */
168     private String assemblySegment(final List<String> cartesianValue, final ReflectValue segment) {
169         String[] segmentAsStringArray = segment.invokeMember("getStrings/()[Ljava/lang/String;").as(String[].class);
170         StringBuilder result = new StringBuilder();
171         for (int i = 0; i < segmentAsStringArray.length; i++) {
172             result.append(segmentAsStringArray[i]);
173             if (i < cartesianValue.size()) {
174                 result.append(cartesianValue.get(i));
175             }
176         }
177         return result.toString();
178     }
179     
180     @Override
181     public String getType() {
182         return "ESPRESSO";
183     }
184 }
