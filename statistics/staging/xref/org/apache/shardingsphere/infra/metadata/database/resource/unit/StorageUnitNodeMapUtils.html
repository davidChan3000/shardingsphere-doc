32  import javax.sql.DataSource;
33  import java.util.LinkedHashMap;
34  import java.util.Map;
35  import java.util.Map.Entry;
36  import java.util.stream.Collectors;
37  
38  /**
39   * Storage unit node map utility class.
40   */
41  @NoArgsConstructor(access = AccessLevel.PRIVATE)
42  public final class StorageUnitNodeMapUtils {
43      
44      /**
45       * Get storage unit node map from data sources.
46       *
47       * @param dataSources data sources
48       * @return storage unit node mappers
49       */
50      public static Map<String, StorageNode> fromDataSources(final Map<String, DataSource> dataSources) {
51          return dataSources.entrySet().stream()
52                  .collect(Collectors.toMap(Entry::getKey, entry -> fromDataSource(entry.getKey(), entry.getValue()), (oldValue, currentValue) -> currentValue, LinkedHashMap::new));
53      }
54      
55      private static StorageNode fromDataSource(final String storageUnitName, final DataSource dataSource) {
56          DataSourcePoolProperties props = DataSourcePoolPropertiesCreator.create(dataSource);
57          String url = props.getConnectionPropertySynonyms().getStandardProperties().get("url").toString();
58          return new StorageNode(new StorageNodeName(storageUnitName), url);
59      }
60      
61      /**
62       * Get storage unit node map from data source pool properties.
63       *
64       * @param propsMap data source pool properties map
65       * @return storage unit node mappers
66       */
67      public static Map<String, StorageNode> fromDataSourcePoolProperties(final Map<String, DataSourcePoolProperties> propsMap) {
68          Map<String, StorageNode> result = new LinkedHashMap<>();
69          for (Entry<String, DataSourcePoolProperties> entry : propsMap.entrySet()) {
70              String storageUnitName = entry.getKey();
71              result.put(storageUnitName, fromDataSourcePoolProperties(storageUnitName, entry.getValue()));
72          }
73          return result;
74      }
75      
76      private static StorageNode fromDataSourcePoolProperties(final String storageUnitName, final DataSourcePoolProperties props) {
77          Map<String, Object> standardProps = props.getConnectionPropertySynonyms().getStandardProperties();
78          String url = standardProps.get("url").toString();
79          boolean isInstanceConnectionAvailable = new DatabaseTypeRegistry(DatabaseTypeFactory.get(url)).getDialectDatabaseMetaData().isInstanceConnectionAvailable();
80          StorageNodeName storageNodeName = getStorageNodeName(storageUnitName, url, standardProps.get("username").toString(), isInstanceConnectionAvailable);
81          return createStorageNode(storageNodeName, url, isInstanceConnectionAvailable);
82      }
83      
84      private static StorageNodeName getStorageNodeName(final String dataSourceName, final String url, final String username, final boolean isInstanceConnectionAvailable) {
85          try {
86              JdbcUrl jdbcUrl = new StandardJdbcUrlParser().parse(url);
87              return isInstanceConnectionAvailable ? new StorageNodeName(jdbcUrl.getHostname(), jdbcUrl.getPort(), username) : new StorageNodeName(dataSourceName);
88          } catch (final UnrecognizedDatabaseURLException ex) {
89              return new StorageNodeName(dataSourceName);
90          }
91      }
92      
93      private static StorageNode createStorageNode(final StorageNodeName storageNodeName, final String url, final boolean isInstanceConnectionAvailable) {
94          return isInstanceConnectionAvailable ? new StorageNode(storageNodeName, url, new StandardJdbcUrlParser().parse(url).getDatabase()) : new StorageNode(storageNodeName, url);
95      }
96  }
