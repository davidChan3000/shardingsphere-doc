39  import javax.sql.DataSource;

44  /**
45   * Pipeline job preparer.
46   */
47  @RequiredArgsConstructor
48  @Slf4j
49  public final class PipelineJobPreparer {
50      
51      private final DatabaseType databaseType;
52      
53      /**
54       * Get incremental position.
55       *
56       * @param initIncremental init incremental
57       * @param dumperContext dumper config
58       * @param dataSourceManager data source manager
59       * @return ingest position
60       * @throws SQLException SQL exception
61       */
62      public IngestPosition getIncrementalPosition(final JobItemIncrementalTasksProgress initIncremental, final IncrementalDumperContext dumperContext,
63                                                   final PipelineDataSourceManager dataSourceManager) throws SQLException {
64          if (null != initIncremental) {
65              Optional<IngestPosition> position = initIncremental.getIncrementalPosition();
66              if (position.isPresent()) {
67                  return position.get();
68              }
69          }
70          DataSource dataSource = dataSourceManager.getDataSource(dumperContext.getCommonContext().getDataSourceConfig());
71          return DatabaseTypedSPILoader.getService(PositionInitializer.class, databaseType).init(dataSource, dumperContext.getJobId());
72      }
73      
74      /**
75       * Check data source.
76       *
77       * @param dataSources data source
78       */
79      public void checkSourceDataSource(final Collection<? extends DataSource> dataSources) {
80          if (dataSources.isEmpty()) {
81              return;
82          }
83          DataSourceCheckEngine checkEngine = new DataSourceCheckEngine(databaseType);
84          checkEngine.checkConnection(dataSources);
85          checkEngine.checkPrivilege(dataSources);
86          checkEngine.checkVariable(dataSources);
87      }
88      
89      /**
90       * Check target data source.
91       *
92       * @param importerConfig importer config
93       * @param targetDataSources target data sources
94       */
95      public void checkTargetDataSource(final ImporterConfiguration importerConfig, final Collection<? extends DataSource> targetDataSources) {
96          if (null == targetDataSources || targetDataSources.isEmpty()) {
97              log.info("target data source is empty, skip check");
98              return;
99          }
100         DataSourceCheckEngine dataSourceCheckEngine = new DataSourceCheckEngine(databaseType);
101         dataSourceCheckEngine.checkConnection(targetDataSources);
102         dataSourceCheckEngine.checkTargetTable(targetDataSources, importerConfig.getTableAndSchemaNameMapper(), importerConfig.getLogicTableNames());
103     }
104     
105     /**
106      * Cleanup job preparer.
107      *
108      * @param jobId pipeline job id
109      * @param pipelineDataSourceConfig pipeline data source config
110      * @throws SQLException SQL exception
111      */
112     public void destroyPosition(final String jobId, final PipelineDataSourceConfiguration pipelineDataSourceConfig) throws SQLException {
113         PositionInitializer positionInitializer = DatabaseTypedSPILoader.getService(PositionInitializer.class, databaseType);
114         final long startTimeMillis = System.currentTimeMillis();
115         log.info("Cleanup database type:{}, data source type:{}", databaseType.getType(), pipelineDataSourceConfig.getType());
116         if (pipelineDataSourceConfig instanceof ShardingSpherePipelineDataSourceConfiguration) {
117             ShardingSpherePipelineDataSourceConfiguration dataSourceConfig = (ShardingSpherePipelineDataSourceConfiguration) pipelineDataSourceConfig;
118             for (DataSourcePoolProperties each : new YamlDataSourceConfigurationSwapper().getDataSourcePoolPropertiesMap(dataSourceConfig.getRootConfig()).values()) {
119                 try (PipelineDataSourceWrapper dataSource = new PipelineDataSourceWrapper(DataSourcePoolCreator.create(each), databaseType)) {
120                     positionInitializer.destroy(dataSource, jobId);
121                 }
122             }
123         }
124         if (pipelineDataSourceConfig instanceof StandardPipelineDataSourceConfiguration) {
125             StandardPipelineDataSourceConfiguration dataSourceConfig = (StandardPipelineDataSourceConfiguration) pipelineDataSourceConfig;
126             try (
127                     PipelineDataSourceWrapper dataSource = new PipelineDataSourceWrapper(
128                             DataSourcePoolCreator.create((DataSourcePoolProperties) dataSourceConfig.getDataSourceConfiguration()), databaseType)) {
129                 positionInitializer.destroy(dataSource, jobId);
130             }
131         }
132         log.info("destroyPosition cost {} ms", System.currentTimeMillis() - startTimeMillis);
133     }
134 }
