110     /**
111      * Create CDC job config.
112      *
113      * @param param create CDC job param
114      * @param sinkType sink type
115      * @param sinkProps sink properties
116      * @return job id
117      */
118     public String createJob(final StreamDataParameter param, final CDCSinkType sinkType, final Properties sinkProps) {
119         PipelineContextKey contextKey = new PipelineContextKey(param.getDatabaseName(), InstanceType.PROXY);
120         YamlCDCJobConfiguration yamlJobConfig = getYamlCDCJobConfiguration(param, sinkType, sinkProps, contextKey);
121         extendYamlJobConfiguration(contextKey, yamlJobConfig);
122         CDCJobConfiguration jobConfig = new YamlCDCJobConfigurationSwapper().swapToObject(yamlJobConfig);
123         ShardingSpherePreconditions.checkState(0 != jobConfig.getJobShardingCount(), () -> new PipelineJobCreationWithInvalidShardingCountException(jobConfig.getJobId()));
124         GovernanceRepositoryAPI repositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobConfig.getJobId()));
125         String jobConfigKey = PipelineMetaDataNode.getJobConfigPath(jobConfig.getJobId());
126         if (repositoryAPI.isExisted(jobConfigKey)) {
127             log.warn("CDC job already exists in registry center, ignore, jobConfigKey={}", jobConfigKey);
128         } else {
129             repositoryAPI.persist(PipelineMetaDataNode.getJobRootPath(jobConfig.getJobId()), getJobClass().getName());
130             JobConfigurationPOJO jobConfigPOJO = jobConfig.convertToJobConfigurationPOJO();
131             jobConfigPOJO.setDisabled(true);
132             repositoryAPI.persist(jobConfigKey, YamlEngine.marshal(jobConfigPOJO));
133             if (!param.isFull()) {
134                 initIncrementalPosition(jobConfig);
135             }
136         }
137         return jobConfig.getJobId();
138     }

140     private YamlCDCJobConfiguration getYamlCDCJobConfiguration(final StreamDataParameter param, final CDCSinkType sinkType, final Properties sinkProps, final PipelineContextKey contextKey) {
141         YamlCDCJobConfiguration result = new YamlCDCJobConfiguration();
142         result.setDatabaseName(param.getDatabaseName());
143         result.setSchemaTableNames(param.getSchemaTableNames());
144         result.setFull(param.isFull());
145         result.setDecodeWithTX(param.isDecodeWithTX());
146         YamlSinkConfiguration sinkConfig = new YamlSinkConfiguration();
147         sinkConfig.setSinkType(sinkType.name());
148         sinkConfig.setProps(sinkProps);
149         result.setSinkConfig(sinkConfig);
150         ShardingSphereDatabase database = PipelineContextManager.getContext(contextKey).getContextManager().getMetaDataContexts().getMetaData().getDatabase(param.getDatabaseName());
151         result.setDataSourceConfiguration(pipelineDataSourceConfigSwapper.swapToYamlConfiguration(getDataSourceConfiguration(database)));
152         List<JobDataNodeLine> jobDataNodeLines = JobDataNodeLineConvertUtils.convertDataNodesToLines(param.getDataNodesMap());
153         result.setJobShardingDataNodes(jobDataNodeLines.stream().map(JobDataNodeLine::marshal).collect(Collectors.toList()));
154         JobDataNodeLine tableFirstDataNodes = new JobDataNodeLine(param.getDataNodesMap().entrySet().stream()
155                 .map(each -> new JobDataNodeEntry(each.getKey(), each.getValue().subList(0, 1))).collect(Collectors.toList()));
156         result.setTablesFirstDataNodes(tableFirstDataNodes.marshal());
157         return result;
158     }

160     private ShardingSpherePipelineDataSourceConfiguration getDataSourceConfiguration(final ShardingSphereDatabase database) {
161         Map<String, Map<String, Object>> dataSourcePoolProps = new HashMap<>();
162         for (Entry<String, StorageUnit> entry : database.getResourceMetaData().getStorageUnits().entrySet()) {
163             dataSourcePoolProps.put(entry.getKey(), dataSourceConfigSwapper.swapToMap(entry.getValue().getDataSourcePoolProperties()));
164         }
165         YamlRootConfiguration targetRootConfig = new YamlRootConfiguration();
166         targetRootConfig.setDatabaseName(database.getName());
167         targetRootConfig.setDataSources(dataSourcePoolProps);
168         targetRootConfig.setRules(ruleConfigSwapperEngine.swapToYamlRuleConfigurations(database.getRuleMetaData().getConfigurations()));
169         return new ShardingSpherePipelineDataSourceConfiguration(targetRootConfig);
170     }

172     private void initIncrementalPosition(final CDCJobConfiguration jobConfig) {
173         String jobId = jobConfig.getJobId();
174         PipelineJobItemManager<InventoryIncrementalJobItemProgress> jobItemManager = new PipelineJobItemManager<>(getYamlJobItemProgressSwapper());
175         try (PipelineDataSourceManager pipelineDataSourceManager = new DefaultPipelineDataSourceManager()) {
176             for (int i = 0; i < jobConfig.getJobShardingCount(); i++) {
177                 if (jobItemManager.getProgress(jobId, i).isPresent()) {
178                     continue;
179                 }
180                 IncrementalDumperContext dumperContext = buildDumperContext(jobConfig, i, new TableAndSchemaNameMapper(jobConfig.getSchemaTableNames()));
181                 InventoryIncrementalJobItemProgress jobItemProgress = getInventoryIncrementalJobItemProgress(jobConfig, pipelineDataSourceManager, dumperContext);
182                 PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).persistJobItemProgress(
183                         jobId, i, YamlEngine.marshal(getYamlJobItemProgressSwapper().swapToYamlConfiguration(jobItemProgress)));
184             }
185         } catch (final SQLException ex) {
186             throw new PrepareJobWithGetBinlogPositionException(jobId, ex);
187         }
188     }
189     
190     private static InventoryIncrementalJobItemProgress getInventoryIncrementalJobItemProgress(final CDCJobConfiguration jobConfig,
191                                                                                               final PipelineDataSourceManager dataSourceManager,
192                                                                                               final IncrementalDumperContext incrementalDumperContext) throws SQLException {
193         InventoryIncrementalJobItemProgress result = new InventoryIncrementalJobItemProgress();
194         result.setSourceDatabaseType(jobConfig.getSourceDatabaseType());
195         result.setDataSourceName(incrementalDumperContext.getCommonContext().getDataSourceName());
196         IncrementalTaskProgress incrementalTaskProgress = new IncrementalTaskProgress(PipelineJobPreparerUtils.getIncrementalPosition(null, incrementalDumperContext, dataSourceManager));
197         result.setIncremental(new JobItemIncrementalTasksProgress(incrementalTaskProgress));
198         return result;
199     }

201     /**
202      * Start job.
203      *
204      * @param jobId job id
205      * @param sink sink
206      */
207     public void startJob(final String jobId, final PipelineSink sink) {
208         CDCJob job = new CDCJob(jobId, sink);
209         PipelineJobCenter.addJob(jobId, job);
210         updateJobConfigurationDisabled(jobId, false);
211         JobConfigurationPOJO jobConfigPOJO = PipelineJobIdUtils.getElasticJobConfigurationPOJO(jobId);
212         OneOffJobBootstrap oneOffJobBootstrap = new OneOffJobBootstrap(PipelineAPIFactory.getRegistryCenter(PipelineJobIdUtils.parseContextKey(jobId)), job, jobConfigPOJO.toJobConfiguration());
213         job.setJobBootstrap(oneOffJobBootstrap);
214         oneOffJobBootstrap.execute();
215     }

217     /**
218      * Update job configuration disabled.
219      *
220      * @param jobId job id
221      * @param disabled disabled
222      */
223     public void updateJobConfigurationDisabled(final String jobId, final boolean disabled) {
224         JobConfigurationPOJO jobConfigPOJO = PipelineJobIdUtils.getElasticJobConfigurationPOJO(jobId);
225         jobConfigPOJO.setDisabled(disabled);
226         if (disabled) {
227             jobConfigPOJO.getProps().setProperty("stop_time", LocalDateTime.now().format(PipelineJobConfiguration.DATE_TIME_FORMATTER));
228             jobConfigPOJO.getProps().setProperty("stop_time_millis", String.valueOf(System.currentTimeMillis()));
229         } else {
230             jobConfigPOJO.getProps().setProperty("start_time_millis", String.valueOf(System.currentTimeMillis()));
231             jobConfigPOJO.getProps().remove("stop_time_millis");
232         }
233         PipelineAPIFactory.getJobConfigurationAPI(PipelineJobIdUtils.parseContextKey(jobConfigPOJO.getJobName())).updateJobConfiguration(jobConfigPOJO);
234     }

236     @Override
237     public void extendYamlJobConfiguration(final PipelineContextKey contextKey, final YamlPipelineJobConfiguration yamlJobConfig) {
238         YamlCDCJobConfiguration config = (YamlCDCJobConfiguration) yamlJobConfig;
239         if (null == yamlJobConfig.getJobId()) {
240             config.setJobId(new CDCJobId(contextKey, config.getSchemaTableNames(), config.isFull(), config.getSinkConfig().getSinkType()).marshal());
241         }
242         if (Strings.isNullOrEmpty(config.getSourceDatabaseType())) {
243             PipelineDataSourceConfiguration sourceDataSourceConfig = PipelineDataSourceConfigurationFactory.newInstance(config.getDataSourceConfiguration().getType(),
244                     config.getDataSourceConfiguration().getParameter());
245             config.setSourceDatabaseType(sourceDataSourceConfig.getDatabaseType().getType());
246         }
247     }

249     @Override
250     public CDCTaskConfiguration buildTaskConfiguration(final PipelineJobConfiguration pipelineJobConfig, final int jobShardingItem, final PipelineProcessConfiguration pipelineProcessConfig) {
251         CDCJobConfiguration jobConfig = (CDCJobConfiguration) pipelineJobConfig;
252         TableAndSchemaNameMapper tableAndSchemaNameMapper = new TableAndSchemaNameMapper(jobConfig.getSchemaTableNames());
253         IncrementalDumperContext dumperContext = buildDumperContext(jobConfig, jobShardingItem, tableAndSchemaNameMapper);
254         ImporterConfiguration importerConfig = buildImporterConfiguration(jobConfig, pipelineProcessConfig, jobConfig.getSchemaTableNames(), tableAndSchemaNameMapper);
255         CDCTaskConfiguration result = new CDCTaskConfiguration(dumperContext, importerConfig);
256         log.debug("buildTaskConfiguration, result={}", result);
257         return result;
258     }
259     
260     private IncrementalDumperContext buildDumperContext(final CDCJobConfiguration jobConfig, final int jobShardingItem, final TableAndSchemaNameMapper tableAndSchemaNameMapper) {
261         JobDataNodeLine dataNodeLine = jobConfig.getJobShardingDataNodes().get(jobShardingItem);
262         String dataSourceName = dataNodeLine.getEntries().iterator().next().getDataNodes().iterator().next().getDataSourceName();
263         StandardPipelineDataSourceConfiguration actualDataSourceConfig = jobConfig.getDataSourceConfig().getActualDataSourceConfiguration(dataSourceName);
264         return new IncrementalDumperContext(
265                 new DumperCommonContext(dataSourceName, actualDataSourceConfig, JobDataNodeLineConvertUtils.buildTableNameMapper(dataNodeLine), tableAndSchemaNameMapper),
266                 jobConfig.getJobId(), jobConfig.isDecodeWithTX());
267     }
268     
269     private ImporterConfiguration buildImporterConfiguration(final CDCJobConfiguration jobConfig, final PipelineProcessConfiguration pipelineProcessConfig, final Collection<String> schemaTableNames,
270                                                              final TableAndSchemaNameMapper tableAndSchemaNameMapper) {
271         PipelineDataSourceConfiguration dataSourceConfig = PipelineDataSourceConfigurationFactory.newInstance(jobConfig.getDataSourceConfig().getType(),
272                 jobConfig.getDataSourceConfig().getParameter());
273         CDCProcessContext processContext = new CDCProcessContext(jobConfig.getJobId(), pipelineProcessConfig);
274         JobRateLimitAlgorithm writeRateLimitAlgorithm = processContext.getWriteRateLimitAlgorithm();
275         int batchSize = pipelineProcessConfig.getWrite().getBatchSize();
276         Map<CaseInsensitiveIdentifier, Set<String>> shardingColumnsMap = new ShardingColumnsExtractor()
277                 .getShardingColumnsMap(jobConfig.getDataSourceConfig().getRootConfig().getRules(), schemaTableNames.stream().map(CaseInsensitiveIdentifier::new).collect(Collectors.toSet()));
278         return new ImporterConfiguration(dataSourceConfig, shardingColumnsMap, tableAndSchemaNameMapper, batchSize, writeRateLimitAlgorithm, 0, 1);
279     }

303     /**
304      * Drop streaming job.
305      *
306      * @param jobId job id
307      */
308     public void dropStreaming(final String jobId) {
309         CDCJobConfiguration jobConfig = new PipelineJobManager(this).getJobConfiguration(jobId);
310         ShardingSpherePreconditions.checkState(PipelineJobIdUtils.getElasticJobConfigurationPOJO(jobId).isDisabled(), () -> new PipelineInternalException("Can't drop streaming job which is active"));
311         new PipelineJobManager(this).drop(jobId);
312         cleanup(jobConfig);
313     }
314     
315     private void cleanup(final CDCJobConfiguration jobConfig) {
316         for (Entry<String, Map<String, Object>> entry : jobConfig.getDataSourceConfig().getRootConfig().getDataSources().entrySet()) {
317             try {
318                 PipelineJobPreparerUtils.destroyPosition(jobConfig.getJobId(), new StandardPipelineDataSourceConfiguration(entry.getValue()));
319             } catch (final SQLException ex) {
320                 log.warn("job destroying failed, jobId={}, dataSourceName={}", jobConfig.getJobId(), entry.getKey(), ex);
321             }
322         }
323     }
