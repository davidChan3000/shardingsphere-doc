78      /**
79       * Create consistency check configuration and start job.
80       *
81       * @param param create consistency check job parameter
82       * @return job id
83       * @throws UncompletedConsistencyCheckJobExistsException uncompleted consistency check job exists exception
84       */
85      public String createJobAndStart(final CreateConsistencyCheckJobParameter param) {
86          String parentJobId = param.getParentJobId();
87          GovernanceRepositoryAPI repositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
88          Optional<String> latestCheckJobId = repositoryAPI.getLatestCheckJobId(parentJobId);
89          if (latestCheckJobId.isPresent()) {
90              Optional<ConsistencyCheckJobItemProgress> progress = getJobItemProgress(latestCheckJobId.get(), 0);
91              if (!progress.isPresent() || JobStatus.FINISHED != progress.get().getStatus()) {
92                  log.info("check job already exists and status is not FINISHED, progress={}", progress);
93                  throw new UncompletedConsistencyCheckJobExistsException(latestCheckJobId.get());
94              }
95          }
96          verifyPipelineDatabaseType(param);
97          PipelineContextKey contextKey = PipelineJobIdUtils.parseContextKey(parentJobId);
98          String result = latestCheckJobId.map(s -> new ConsistencyCheckJobId(contextKey, parentJobId, s)).orElseGet(() -> new ConsistencyCheckJobId(contextKey, parentJobId)).marshal();
99          repositoryAPI.persistLatestCheckJobId(parentJobId, result);
100         repositoryAPI.deleteCheckJobResult(parentJobId, result);
101         new PipelineJobManager(this).drop(result);
102         YamlConsistencyCheckJobConfiguration yamlConfig = new YamlConsistencyCheckJobConfiguration();
103         yamlConfig.setJobId(result);
104         yamlConfig.setParentJobId(parentJobId);
105         yamlConfig.setAlgorithmTypeName(param.getAlgorithmTypeName());
106         yamlConfig.setAlgorithmProps(param.getAlgorithmProps());
107         yamlConfig.setSourceDatabaseType(param.getSourceDatabaseType().getType());
108         new PipelineJobManager(this).start(new YamlConsistencyCheckJobConfigurationSwapper().swapToObject(yamlConfig));
109         return result;
110     }
111     
112     private void verifyPipelineDatabaseType(final CreateConsistencyCheckJobParameter param) {
113         Collection<DatabaseType> supportedDatabaseTypes = TableDataConsistencyCheckerFactory.newInstance(param.getAlgorithmTypeName(), param.getAlgorithmProps()).getSupportedDatabaseTypes();
114         ShardingSpherePreconditions.checkState(supportedDatabaseTypes.contains(param.getSourceDatabaseType()), () -> new UnsupportedPipelineDatabaseTypeException(param.getSourceDatabaseType()));
115         ShardingSpherePreconditions.checkState(supportedDatabaseTypes.contains(param.getTargetDatabaseType()), () -> new UnsupportedPipelineDatabaseTypeException(param.getTargetDatabaseType()));
116     }

141         Optional<String> progress = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).getJobItemProgress(jobId, shardingItem);

145     @Override
146     public void updateJobItemStatus(final String jobId, final int shardingItem, final JobStatus status) {
147         Optional<ConsistencyCheckJobItemProgress> jobItemProgress = getJobItemProgress(jobId, shardingItem);
148         if (!jobItemProgress.isPresent()) {
149             log.warn("updateJobItemStatus, jobProgress is null, jobId={}, shardingItem={}", jobId, shardingItem);
150             return;
151         }
152         jobItemProgress.get().setStatus(status);
153         PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(jobId)).updateJobItemProgress(jobId, shardingItem,
154                 YamlEngine.marshal(swapper.swapToYamlConfiguration(jobItemProgress.get())));
155     }

166     private String getLatestCheckJobId(final String parentJobId) {
167         Optional<String> result = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId)).getLatestCheckJobId(parentJobId);
168         ShardingSpherePreconditions.checkState(result.isPresent(), () -> new ConsistencyCheckJobNotFoundException(parentJobId));

181     /**
182      * Drop by parent job id.
183      *
184      * @param parentJobId parent job id
185      */
186     public void dropByParentJobId(final String parentJobId) {
187         String latestCheckJobId = getLatestCheckJobId(parentJobId);
188         new PipelineJobManager(this).stop(latestCheckJobId);
189         PipelineContextKey contextKey = PipelineJobIdUtils.parseContextKey(parentJobId);
190         GovernanceRepositoryAPI repositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(contextKey);
191         Collection<String> checkJobIds = repositoryAPI.listCheckJobIds(parentJobId);
192         Optional<Integer> previousSequence = ConsistencyCheckSequence.getPreviousSequence(
193                 checkJobIds.stream().map(ConsistencyCheckJobId::parseSequence).collect(Collectors.toList()), ConsistencyCheckJobId.parseSequence(latestCheckJobId));
194         if (previousSequence.isPresent()) {
195             String checkJobId = new ConsistencyCheckJobId(contextKey, parentJobId, previousSequence.get()).marshal();
196             repositoryAPI.persistLatestCheckJobId(parentJobId, checkJobId);
197         } else {
198             repositoryAPI.deleteLatestCheckJobId(parentJobId);
199         }
200         repositoryAPI.deleteCheckJobResult(parentJobId, latestCheckJobId);
201         new PipelineJobManager(this).drop(latestCheckJobId);
202     }

204     /**
205      * Get consistency job item infos.
206      *
207      * @param parentJobId parent job id
208      * @return consistency job item infos
209      */
210     public List<ConsistencyCheckJobItemInfo> getJobItemInfos(final String parentJobId) {
211         GovernanceRepositoryAPI governanceRepositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
212         Optional<String> latestCheckJobId = governanceRepositoryAPI.getLatestCheckJobId(parentJobId);
213         ShardingSpherePreconditions.checkState(latestCheckJobId.isPresent(), () -> new ConsistencyCheckJobNotFoundException(parentJobId));
214         String checkJobId = latestCheckJobId.get();
215         Optional<ConsistencyCheckJobItemProgress> progress = getJobItemProgress(checkJobId, 0);
216         if (!progress.isPresent()) {
217             return Collections.emptyList();
218         }
219         List<ConsistencyCheckJobItemInfo> result = new LinkedList<>();
220         ConsistencyCheckJobItemProgress jobItemProgress = progress.get();
221         if (!Strings.isNullOrEmpty(jobItemProgress.getIgnoredTableNames())) {
222             Map<String, TableDataConsistencyCheckResult> checkJobResult = governanceRepositoryAPI.getCheckJobResult(parentJobId, latestCheckJobId.get());
223             result.addAll(buildIgnoredTableInfo(jobItemProgress.getIgnoredTableNames().split(","), checkJobResult));
224         }
225         if (Objects.equals(jobItemProgress.getIgnoredTableNames(), jobItemProgress.getTableNames())) {
226             return result;
227         }
228         result.add(getJobItemInfo(parentJobId));
229         return result;
230     }
231     
232     private List<ConsistencyCheckJobItemInfo> buildIgnoredTableInfo(final String[] ignoredTables, final Map<String, TableDataConsistencyCheckResult> checkJobResult) {
233         if (null == ignoredTables) {
234             return Collections.emptyList();
235         }
236         List<ConsistencyCheckJobItemInfo> result = new LinkedList<>();
237         for (String each : ignoredTables) {
238             ConsistencyCheckJobItemInfo info = new ConsistencyCheckJobItemInfo();
239             info.setTableNames(each);
240             info.setCheckSuccess(null);
241             TableDataConsistencyCheckResult checkResult = checkJobResult.get(each);
242             if (null != checkResult && checkResult.isIgnored()) {
243                 info.setErrorMessage(checkResult.getIgnoredType().getMessage());
244             }
245             result.add(info);
246         }
247         return result;
248     }

250     private ConsistencyCheckJobItemInfo getJobItemInfo(final String parentJobId) {
251         GovernanceRepositoryAPI governanceRepositoryAPI = PipelineAPIFactory.getGovernanceRepositoryAPI(PipelineJobIdUtils.parseContextKey(parentJobId));
252         Optional<String> latestCheckJobId = governanceRepositoryAPI.getLatestCheckJobId(parentJobId);
253         ShardingSpherePreconditions.checkState(latestCheckJobId.isPresent(), () -> new ConsistencyCheckJobNotFoundException(parentJobId));
254         String checkJobId = latestCheckJobId.get();
255         Optional<ConsistencyCheckJobItemProgress> progress = getJobItemProgress(checkJobId, 0);
256         ConsistencyCheckJobItemInfo result = new ConsistencyCheckJobItemInfo();
257         JobConfigurationPOJO jobConfigPOJO = PipelineJobIdUtils.getElasticJobConfigurationPOJO(checkJobId);
258         result.setActive(!jobConfigPOJO.isDisabled());
259         if (!progress.isPresent()) {
260             return result;
261         }
262         ConsistencyCheckJobItemProgress jobItemProgress = progress.get();
263         if (null == jobItemProgress.getRecordsCount() || null == jobItemProgress.getCheckedRecordsCount()) {
264             result.setInventoryFinishedPercentage(0);
265             result.setCheckSuccess(null);
266             return result;
267         }
268         fillInJobItemInfoWithTimes(result, jobItemProgress, jobConfigPOJO);
269         result.setTableNames(Optional.ofNullable(jobItemProgress.getTableNames()).orElse(""));
270         fillInJobItemInfoWithCheckAlgorithm(result, checkJobId);
271         result.setErrorMessage(new PipelineJobManager(this).getJobItemErrorMessage(checkJobId, 0));
272         Map<String, TableDataConsistencyCheckResult> checkJobResult = governanceRepositoryAPI.getCheckJobResult(parentJobId, checkJobId);
273         fillInJobItemInfoWithCheckResult(result, checkJobResult, parentJobId);
274         result.setCheckFailedTableNames(checkJobResult.entrySet().stream().filter(each -> !each.getValue().isIgnored() && !each.getValue().isMatched())
275                 .map(Entry::getKey).collect(Collectors.joining(",")));
276         return result;
277     }
278     
279     private void fillInJobItemInfoWithTimes(final ConsistencyCheckJobItemInfo result, final ConsistencyCheckJobItemProgress jobItemProgress, final JobConfigurationPOJO jobConfigPOJO) {
280         long recordsCount = jobItemProgress.getRecordsCount();
281         long checkedRecordsCount = Math.min(jobItemProgress.getCheckedRecordsCount(), recordsCount);
282         LocalDateTime checkBeginTime = new Timestamp(jobItemProgress.getCheckBeginTimeMillis()).toLocalDateTime();
283         result.setCheckBeginTime(DATE_TIME_FORMATTER.format(checkBeginTime));
284         if (JobStatus.FINISHED == jobItemProgress.getStatus()) {
285             result.setInventoryFinishedPercentage(100);
286             LocalDateTime checkEndTime = new Timestamp(jobItemProgress.getCheckEndTimeMillis()).toLocalDateTime();
287             Duration duration = Duration.between(checkBeginTime, checkEndTime);
288             result.setDurationSeconds(duration.getSeconds());
289             result.setCheckEndTime(DATE_TIME_FORMATTER.format(checkEndTime));
290             result.setInventoryRemainingSeconds(0L);
291         } else if (0 != recordsCount && 0 != checkedRecordsCount) {
292             result.setInventoryFinishedPercentage((int) (checkedRecordsCount * 100 / recordsCount));
293             Long stopTimeMillis = jobConfigPOJO.isDisabled() ? Long.parseLong(jobConfigPOJO.getProps().getProperty("stop_time_millis")) : null;
294             long durationMillis = (null != stopTimeMillis ? stopTimeMillis : System.currentTimeMillis()) - jobItemProgress.getCheckBeginTimeMillis();
295             result.setDurationSeconds(TimeUnit.MILLISECONDS.toSeconds(durationMillis));
296             if (null != stopTimeMillis) {
297                 result.setCheckEndTime(DATE_TIME_FORMATTER.format(new Timestamp(stopTimeMillis).toLocalDateTime()));
298             }
299             long remainingMills = Math.max(0, (long) ((recordsCount - checkedRecordsCount) * 1.0D / checkedRecordsCount * durationMillis));
300             result.setInventoryRemainingSeconds(remainingMills / 1000);
301         }
302     }

304     private void fillInJobItemInfoWithCheckAlgorithm(final ConsistencyCheckJobItemInfo result, final String checkJobId) {
305         ConsistencyCheckJobConfiguration jobConfig = (ConsistencyCheckJobConfiguration) new PipelineJobManager(this)
306                 .getJobConfiguration(PipelineJobIdUtils.getElasticJobConfigurationPOJO(checkJobId));
307         result.setAlgorithmType(jobConfig.getAlgorithmTypeName());
308         if (null != jobConfig.getAlgorithmProps()) {
309             result.setAlgorithmProps(jobConfig.getAlgorithmProps().entrySet().stream().map(entry -> String.format("'%s'='%s'", entry.getKey(), entry.getValue())).collect(Collectors.joining(",")));
310         }
311     }
312     
313     private void fillInJobItemInfoWithCheckResult(final ConsistencyCheckJobItemInfo result, final Map<String, TableDataConsistencyCheckResult> checkJobResult, final String parentJobId) {
314         if (checkJobResult.isEmpty()) {
315             result.setCheckSuccess(null);
316         } else {
317             InventoryIncrementalJobAPI inventoryIncrementalJobAPI = (InventoryIncrementalJobAPI) TypedSPILoader.getService(
318                     PipelineJobAPI.class, PipelineJobIdUtils.parseJobType(parentJobId).getType());
319             result.setCheckSuccess(inventoryIncrementalJobAPI.aggregateDataConsistencyCheckResults(parentJobId, checkJobResult));
320         }
321     }
