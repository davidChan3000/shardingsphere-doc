85      public MigrationJobManager(final MigrationJobOption jobOption) {
86          this.jobOption = jobOption;
87          jobManager = new PipelineJobManager(jobOption);
88          dataSourcePersistService = new PipelineDataSourcePersistService();
89      }

91      /**
92       * Start migration job.
93       *
94       * @param contextKey context key
95       * @param param create migration job parameter
96       * @return job id
97       */
98      public String start(final PipelineContextKey contextKey, final MigrateTableStatement param) {
99          MigrationJobConfiguration jobConfig = new YamlMigrationJobConfigurationSwapper().swapToObject(buildYamlJobConfiguration(contextKey, param));
100         jobManager.start(jobConfig);
101         return jobConfig.getJobId();
102     }
103     
104     private YamlMigrationJobConfiguration buildYamlJobConfiguration(final PipelineContextKey contextKey, final MigrateTableStatement param) {
105         YamlMigrationJobConfiguration result = new YamlMigrationJobConfiguration();
106         result.setTargetDatabaseName(param.getTargetDatabaseName());
107         Map<String, DataSourcePoolProperties> metaDataDataSource = dataSourcePersistService.load(contextKey, "MIGRATION");
108         Map<String, List<DataNode>> sourceDataNodes = new LinkedHashMap<>();
109         Map<String, YamlPipelineDataSourceConfiguration> configSources = new LinkedHashMap<>();
110         List<SourceTargetEntry> sourceTargetEntries = new ArrayList<>(new HashSet<>(param.getSourceTargetEntries())).stream().sorted(Comparator.comparing(SourceTargetEntry::getTargetTableName)
111                 .thenComparing(each -> DataNodeUtils.formatWithSchema(each.getSource()))).collect(Collectors.toList());
112         YamlDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlDataSourceConfigurationSwapper();
113         for (SourceTargetEntry each : sourceTargetEntries) {
114             sourceDataNodes.computeIfAbsent(each.getTargetTableName(), key -> new LinkedList<>()).add(each.getSource());
115             ShardingSpherePreconditions.checkState(1 == sourceDataNodes.get(each.getTargetTableName()).size(),
116                     () -> new PipelineInvalidParameterException("more than one source table for " + each.getTargetTableName()));
117             String dataSourceName = each.getSource().getDataSourceName();
118             if (configSources.containsKey(dataSourceName)) {
119                 continue;
120             }
121             ShardingSpherePreconditions.checkState(metaDataDataSource.containsKey(dataSourceName),
122                     () -> new PipelineInvalidParameterException(dataSourceName + " doesn't exist. Run `SHOW MIGRATION SOURCE STORAGE UNITS;` to verify it."));
123             Map<String, Object> sourceDataSourcePoolProps = dataSourceConfigSwapper.swapToMap(metaDataDataSource.get(dataSourceName));
124             StandardPipelineDataSourceConfiguration sourceDataSourceConfig = new StandardPipelineDataSourceConfiguration(sourceDataSourcePoolProps);
125             configSources.put(dataSourceName, buildYamlPipelineDataSourceConfiguration(sourceDataSourceConfig.getType(), sourceDataSourceConfig.getParameter()));
126             DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(sourceDataSourceConfig.getDatabaseType()).getDialectDatabaseMetaData();
127             if (null == each.getSource().getSchemaName() && dialectDatabaseMetaData.isSchemaAvailable()) {
128                 each.getSource().setSchemaName(PipelineSchemaUtils.getDefaultSchema(sourceDataSourceConfig));
129             }
130             DatabaseType sourceDatabaseType = sourceDataSourceConfig.getDatabaseType();
131             if (null == result.getSourceDatabaseType()) {
132                 result.setSourceDatabaseType(sourceDatabaseType.getType());
133             } else if (!result.getSourceDatabaseType().equals(sourceDatabaseType.getType())) {
134                 throw new PipelineInvalidParameterException("Source storage units have different database types");
135             }
136         }
137         result.setSources(configSources);
138         ShardingSphereDatabase targetDatabase = PipelineContextManager.getProxyContext().getContextManager().getMetaDataContexts().getMetaData().getDatabase(param.getTargetDatabaseName());
139         PipelineDataSourceConfiguration targetPipelineDataSourceConfig = buildTargetPipelineDataSourceConfiguration(targetDatabase);
140         result.setTarget(buildYamlPipelineDataSourceConfiguration(targetPipelineDataSourceConfig.getType(), targetPipelineDataSourceConfig.getParameter()));
141         result.setTargetDatabaseType(targetPipelineDataSourceConfig.getDatabaseType().getType());
142         List<JobDataNodeEntry> tablesFirstDataNodes = sourceDataNodes.entrySet().stream()
143                 .map(entry -> new JobDataNodeEntry(entry.getKey(), entry.getValue().subList(0, 1))).collect(Collectors.toList());
144         result.setTargetTableNames(new ArrayList<>(sourceDataNodes.keySet()).stream().sorted().collect(Collectors.toList()));
145         result.setTargetTableSchemaMap(buildTargetTableSchemaMap(sourceDataNodes));
146         result.setTablesFirstDataNodes(new JobDataNodeLine(tablesFirstDataNodes).marshal());
147         result.setJobShardingDataNodes(JobDataNodeLineConvertUtils.convertDataNodesToLines(sourceDataNodes).stream().map(JobDataNodeLine::marshal).collect(Collectors.toList()));
148         jobOption.extendYamlJobConfiguration(contextKey, result);
149         return result;
150     }
151     
152     private YamlPipelineDataSourceConfiguration buildYamlPipelineDataSourceConfiguration(final String type, final String param) {
153         YamlPipelineDataSourceConfiguration result = new YamlPipelineDataSourceConfiguration();
154         result.setType(type);
155         result.setParameter(param);
156         return result;
157     }
158     
159     private PipelineDataSourceConfiguration buildTargetPipelineDataSourceConfiguration(final ShardingSphereDatabase targetDatabase) {
160         Map<String, Map<String, Object>> targetPoolProps = new HashMap<>();
161         YamlDataSourceConfigurationSwapper dataSourceConfigSwapper = new YamlDataSourceConfigurationSwapper();
162         for (Entry<String, StorageUnit> entry : targetDatabase.getResourceMetaData().getStorageUnits().entrySet()) {
163             targetPoolProps.put(entry.getKey(), dataSourceConfigSwapper.swapToMap(entry.getValue().getDataSourcePoolProperties()));
164         }
165         YamlRootConfiguration targetRootConfig = buildYamlRootConfiguration(targetDatabase.getName(), targetPoolProps, targetDatabase.getRuleMetaData().getConfigurations());
166         return new ShardingSpherePipelineDataSourceConfiguration(targetRootConfig);
167     }
168     
169     private YamlRootConfiguration buildYamlRootConfiguration(final String databaseName, final Map<String, Map<String, Object>> yamlDataSources, final Collection<RuleConfiguration> rules) {
170         ShardingSpherePreconditions.checkState(!rules.isEmpty(), () -> new NoAnyRuleExistsException(databaseName));
171         YamlRootConfiguration result = new YamlRootConfiguration();
172         result.setDatabaseName(databaseName);
173         result.setDataSources(yamlDataSources);
174         result.setRules(new YamlRuleConfigurationSwapperEngine().swapToYamlRuleConfigurations(rules));
175         return result;
176     }
177     
178     private Map<String, String> buildTargetTableSchemaMap(final Map<String, List<DataNode>> sourceDataNodes) {
179         Map<String, String> result = new LinkedHashMap<>();
180         sourceDataNodes.forEach((tableName, dataNodes) -> result.put(tableName, dataNodes.get(0).getSchemaName()));
181         return result;
182     }

184     /**
185      * Add migration source resources.
186      *
187      * @param contextKey context key
188      * @param propsMap data source pool properties map
189      */
190     public void addMigrationSourceResources(final PipelineContextKey contextKey, final Map<String, DataSourcePoolProperties> propsMap) {
191         Map<String, DataSourcePoolProperties> existDataSources = dataSourcePersistService.load(contextKey, jobOption.getType());
192         Collection<String> duplicateDataSourceNames = new HashSet<>(propsMap.size(), 1F);
193         for (Entry<String, DataSourcePoolProperties> entry : propsMap.entrySet()) {
194             if (existDataSources.containsKey(entry.getKey())) {
195                 duplicateDataSourceNames.add(entry.getKey());
196             }
197         }
198         ShardingSpherePreconditions.checkState(duplicateDataSourceNames.isEmpty(), () -> new RegisterMigrationSourceStorageUnitException(duplicateDataSourceNames));
199         Map<String, DataSourcePoolProperties> result = new LinkedHashMap<>(existDataSources);
200         result.putAll(propsMap);
201         dataSourcePersistService.persist(contextKey, jobOption.getType(), result);
202     }
203     
204     /**
205      * Drop migration source resources.
206      *
207      * @param contextKey context key
208      * @param resourceNames resource names
209      */
210     public void dropMigrationSourceResources(final PipelineContextKey contextKey, final Collection<String> resourceNames) {
211         Map<String, DataSourcePoolProperties> metaDataDataSource = dataSourcePersistService.load(contextKey, jobOption.getType());
212         List<String> noExistResources = resourceNames.stream().filter(each -> !metaDataDataSource.containsKey(each)).collect(Collectors.toList());
213         ShardingSpherePreconditions.checkState(noExistResources.isEmpty(), () -> new UnregisterMigrationSourceStorageUnitException(noExistResources));
214         for (String each : resourceNames) {
215             metaDataDataSource.remove(each);
216         }
217         dataSourcePersistService.persist(contextKey, jobOption.getType(), metaDataDataSource);
218     }

220     /**
221      * Query migration source resources list.
222      *
223      * @param contextKey context key
224      * @return migration source resources
225      */
226     public Collection<Collection<Object>> listMigrationSourceResources(final PipelineContextKey contextKey) {
227         Map<String, DataSourcePoolProperties> propsMap = dataSourcePersistService.load(contextKey, jobOption.getType());
228         Collection<Collection<Object>> result = new ArrayList<>(propsMap.size());
229         for (Entry<String, DataSourcePoolProperties> entry : propsMap.entrySet()) {
230             String dataSourceName = entry.getKey();
231             DataSourcePoolProperties value = entry.getValue();
232             Collection<Object> props = new LinkedList<>();
233             props.add(dataSourceName);
234             String url = String.valueOf(value.getConnectionPropertySynonyms().getStandardProperties().get("url"));
235             DatabaseType databaseType = DatabaseTypeFactory.get(url);
236             props.add(databaseType.getType());
237             ConnectionProperties connectionProps = DatabaseTypedSPILoader.getService(ConnectionPropertiesParser.class, databaseType).parse(url, "", null);
238             props.add(connectionProps.getHostname());
239             props.add(connectionProps.getPort());
240             props.add(connectionProps.getCatalog());
241             Map<String, Object> standardProps = value.getPoolPropertySynonyms().getStandardProperties();
242             props.add(getStandardProperty(standardProps, "connectionTimeoutMilliseconds"));
243             props.add(getStandardProperty(standardProps, "idleTimeoutMilliseconds"));
244             props.add(getStandardProperty(standardProps, "maxLifetimeMilliseconds"));
245             props.add(getStandardProperty(standardProps, "maxPoolSize"));
246             props.add(getStandardProperty(standardProps, "minPoolSize"));
247             props.add(getStandardProperty(standardProps, "readOnly"));
248             Map<String, Object> otherProps = value.getCustomProperties().getProperties();
249             props.add(otherProps.isEmpty() ? "" : JsonUtils.toJsonString(otherProps));
250             result.add(props);
251         }
252         return result;
253     }
254     
255     private String getStandardProperty(final Map<String, Object> standardProps, final String key) {
256         if (standardProps.containsKey(key) && null != standardProps.get(key)) {
257             return standardProps.get(key).toString();
258         }
259         return "";
260     }
