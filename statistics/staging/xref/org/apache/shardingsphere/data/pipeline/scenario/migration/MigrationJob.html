65  /**
66   * Migration job.
67   */
68  @Slf4j
69  public final class MigrationJob extends AbstractSimplePipelineJob {
70      
71      private final PipelineJobItemManager<TransmissionJobItemProgress> jobItemManager = new PipelineJobItemManager<>(new MigrationJobType().getYamlJobItemProgressSwapper());
72      
73      private final PipelineProcessConfigurationPersistService processConfigPersistService = new PipelineProcessConfigurationPersistService();
74      
75      private final PipelineDataSourceManager dataSourceManager = new DefaultPipelineDataSourceManager();
76      
77      // Shared by all sharding items
78      private final MigrationJobPreparer jobPreparer = new MigrationJobPreparer();
79      
80      public MigrationJob(final String jobId) {
81          super(jobId);
82      }
83      
84      @Override
85      protected TransmissionJobItemContext buildPipelineJobItemContext(final ShardingContext shardingContext) {
86          int shardingItem = shardingContext.getShardingItem();
87          MigrationJobConfiguration jobConfig = new YamlMigrationJobConfigurationSwapper().swapToObject(shardingContext.getJobParameter());
88          Optional<TransmissionJobItemProgress> initProgress = jobItemManager.getProgress(shardingContext.getJobName(), shardingItem);
89          PipelineProcessConfiguration processConfig = PipelineProcessConfigurationUtils.convertWithDefaultValue(
90                  processConfigPersistService.load(PipelineJobIdUtils.parseContextKey(jobConfig.getJobId()), "MIGRATION"));
91          TransmissionProcessContext jobProcessContext = new TransmissionProcessContext(jobConfig.getJobId(), processConfig);
92          MigrationTaskConfiguration taskConfig = buildTaskConfiguration(jobConfig, shardingItem, jobProcessContext.getPipelineProcessConfig());
93          return new MigrationJobItemContext(jobConfig, shardingItem, initProgress.orElse(null), jobProcessContext, taskConfig, dataSourceManager);
94      }
95      
96      private MigrationTaskConfiguration buildTaskConfiguration(final MigrationJobConfiguration jobConfig, final int jobShardingItem, final PipelineProcessConfiguration processConfig) {
97          IncrementalDumperContext incrementalDumperContext = new MigrationIncrementalDumperContextCreator(jobConfig).createDumperContext(jobConfig.getJobShardingDataNodes().get(jobShardingItem));
98          Collection<CreateTableConfiguration> createTableConfigs = buildCreateTableConfigurations(jobConfig, incrementalDumperContext.getCommonContext().getTableAndSchemaNameMapper());
99          Set<CaseInsensitiveIdentifier> targetTableNames = jobConfig.getTargetTableNames().stream().map(CaseInsensitiveIdentifier::new).collect(Collectors.toSet());
100         Map<CaseInsensitiveIdentifier, Set<String>> shardingColumnsMap = new ShardingColumnsExtractor().getShardingColumnsMap(
101                 ((ShardingSpherePipelineDataSourceConfiguration) jobConfig.getTarget()).getRootConfig().getRules(), targetTableNames);
102         ImporterConfiguration importerConfig = buildImporterConfiguration(jobConfig, processConfig, shardingColumnsMap, incrementalDumperContext.getCommonContext().getTableAndSchemaNameMapper());
103         MigrationTaskConfiguration result = new MigrationTaskConfiguration(
104                 incrementalDumperContext.getCommonContext().getDataSourceName(), createTableConfigs, incrementalDumperContext, importerConfig);
105         log.info("buildTaskConfiguration, result={}", result);
106         return result;
107     }
108     
109     private Collection<CreateTableConfiguration> buildCreateTableConfigurations(final MigrationJobConfiguration jobConfig, final TableAndSchemaNameMapper tableAndSchemaNameMapper) {
110         Collection<CreateTableConfiguration> result = new LinkedList<>();
111         for (JobDataNodeEntry each : jobConfig.getTablesFirstDataNodes().getEntries()) {
112             String sourceSchemaName = tableAndSchemaNameMapper.getSchemaName(each.getLogicTableName());
113             DialectDatabaseMetaData dialectDatabaseMetaData = new DatabaseTypeRegistry(jobConfig.getTargetDatabaseType()).getDialectDatabaseMetaData();
114             String targetSchemaName = dialectDatabaseMetaData.isSchemaAvailable() ? sourceSchemaName : null;
115             DataNode dataNode = each.getDataNodes().get(0);
116             PipelineDataSourceConfiguration sourceDataSourceConfig = jobConfig.getSources().get(dataNode.getDataSourceName());
117             CreateTableConfiguration createTableConfig = new CreateTableConfiguration(sourceDataSourceConfig, new CaseInsensitiveQualifiedTable(sourceSchemaName, dataNode.getTableName()),
118                     jobConfig.getTarget(), new CaseInsensitiveQualifiedTable(targetSchemaName, each.getLogicTableName()));
119             result.add(createTableConfig);
120         }
121         log.info("buildCreateTableConfigurations, result={}", result);
122         return result;
123     }
124     
125     private ImporterConfiguration buildImporterConfiguration(final MigrationJobConfiguration jobConfig, final PipelineProcessConfiguration pipelineProcessConfig,
126                                                              final Map<CaseInsensitiveIdentifier, Set<String>> shardingColumnsMap, final TableAndSchemaNameMapper tableAndSchemaNameMapper) {
127         int batchSize = pipelineProcessConfig.getWrite().getBatchSize();
128         JobRateLimitAlgorithm writeRateLimitAlgorithm = new TransmissionProcessContext(jobConfig.getJobId(), pipelineProcessConfig).getWriteRateLimitAlgorithm();
129         int retryTimes = jobConfig.getRetryTimes();
130         int concurrency = jobConfig.getConcurrency();
131         return new ImporterConfiguration(jobConfig.getTarget(), shardingColumnsMap, tableAndSchemaNameMapper, batchSize, writeRateLimitAlgorithm, retryTimes, concurrency);
132     }
133     
134     @Override
135     protected PipelineTasksRunner buildPipelineTasksRunner(final PipelineJobItemContext pipelineJobItemContext) {
136         return new TransmissionTasksRunner((TransmissionJobItemContext) pipelineJobItemContext);
137     }
138     
139     @Override
140     protected void doPrepare(final PipelineJobItemContext jobItemContext) throws SQLException {
141         jobPreparer.prepare((MigrationJobItemContext) jobItemContext);
142     }
143     
144     @Override
145     public void doClean() {
146         dataSourceManager.close();
147     }
