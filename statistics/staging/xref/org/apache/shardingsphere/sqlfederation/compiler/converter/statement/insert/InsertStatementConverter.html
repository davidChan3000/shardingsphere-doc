51  /**
52   * Insert statement converter.
53   */
54  public final class InsertStatementConverter implements SQLStatementConverter<InsertStatement, SqlNode> {
55      
56      @Override
57      public SqlNode convert(final InsertStatement insertStatement) {
58          return convertInsert(insertStatement);
59      }
60      
61      private SqlInsert convertInsert(final InsertStatement insertStatement) {
62          SqlNode table = new TableConverter().convert(insertStatement.getTable()).orElseThrow(IllegalStateException::new);
63          SqlParserPos position = SqlParserPos.ZERO;
64          SqlNodeList keywords = new SqlNodeList(position);
65          SqlNode source;
66          if (insertStatement.getInsertSelect().isPresent()) {
67              source = convertSelect(insertStatement.getInsertSelect().get());
68          } else {
69              source = convertValues(insertStatement.getValues());
70          }
71          SqlNodeList columnList = convertColumn(insertStatement.getColumns());
72          return new SqlInsert(SqlParserPos.ZERO, keywords, table, source, columnList);
73      }
74      
75      private SqlNode convertSelect(final SubquerySegment subquerySegment) {
76          SelectStatement selectStatement = subquerySegment.getSelect();
77          SqlNodeList distinct = new DistinctConverter().convert(selectStatement.getProjections()).orElse(null);
78          SqlNodeList projection = new ProjectionsConverter().convert(selectStatement.getProjections()).orElseThrow(IllegalStateException::new);
79          SqlNode from = new TableConverter().convert(selectStatement.getFrom()).orElse(null);
80          SqlNode where = selectStatement.getWhere().flatMap(optional -> new WhereConverter().convert(optional)).orElse(null);
81          SqlNodeList groupBy = selectStatement.getGroupBy().flatMap(optional -> new GroupByConverter().convert(optional)).orElse(null);
82          SqlNode having = selectStatement.getHaving().flatMap(optional -> new HavingConverter().convert(optional)).orElse(null);
83          SqlNodeList window = SelectStatementHandler.getWindowSegment(selectStatement).flatMap(new WindowConverter()::convert).orElse(SqlNodeList.EMPTY);
84          return new SqlSelect(SqlParserPos.ZERO, distinct, projection, from, where, groupBy, having, window, null, null, null, null, SqlNodeList.EMPTY);
85      }
86      
87      private SqlNode convertValues(final Collection<InsertValuesSegment> insertValuesSegments) {
88          List<SqlNode> values = new ArrayList<>();
89          for (InsertValuesSegment each : insertValuesSegments) {
90              values.add(convertExpression(each.getValues().get(0)));
91          }
92          List<SqlNode> operands = new ArrayList<>();
93          operands.add(new SqlBasicCall(new SqlRowOperator("ROW"), values, SqlParserPos.ZERO));
94          return new SqlBasicCall(new SqlValuesOperator(), operands, SqlParserPos.ZERO);
95      }
96      
97      private SqlNodeList convertColumn(final Collection<ColumnSegment> columnSegments) {
98          List<SqlNode> columns = columnSegments.stream().map(each -> new ColumnConverter().convert(each).orElseThrow(IllegalStateException::new)).collect(Collectors.toList());
99          if (columns.isEmpty()) {
100             return null;
101         }
102         return new SqlNodeList(columns, SqlParserPos.ZERO);
103     }
104     
105     private SqlNode convertExpression(final ExpressionSegment expressionSegment) {
106         return new ExpressionConverter().convert(expressionSegment).orElseThrow(IllegalStateException::new);
107     }
108 }
