43  /**
44   * Subquery table segment binder.
45   */
46  @NoArgsConstructor(access = AccessLevel.PRIVATE)
47  public final class SubqueryTableSegmentBinder {
48      
49      /**
50       * Bind subquery table segment with metadata.
51       *
52       * @param segment join table segment
53       * @param statementBinderContext statement binder context
54       * @param tableBinderContexts table binder contexts
55       * @param outerTableBinderContexts outer table binder contexts
56       * @return bounded subquery table segment
57       */
58      public static SubqueryTableSegment bind(final SubqueryTableSegment segment, final SQLStatementBinderContext statementBinderContext,
59                                              final Map<String, TableSegmentBinderContext> tableBinderContexts, final Map<String, TableSegmentBinderContext> outerTableBinderContexts) {
60          fillPivotColumnNamesInBinderContext(segment, statementBinderContext);
61          SelectStatement boundedSelect = new SelectStatementBinder().bindCorrelateSubquery(segment.getSubquery().getSelect(), statementBinderContext.getMetaData(),
62                  statementBinderContext.getDefaultDatabaseName(), outerTableBinderContexts, statementBinderContext.getExternalTableBinderContexts());
63          SubquerySegment boundedSubquerySegment = new SubquerySegment(segment.getSubquery().getStartIndex(), segment.getSubquery().getStopIndex(), boundedSelect);
64          boundedSubquerySegment.setSubqueryType(segment.getSubquery().getSubqueryType());
65          SubqueryTableSegment result = new SubqueryTableSegment(boundedSubquerySegment);
66          segment.getAliasSegment().ifPresent(result::setAlias);
67          IdentifierValue subqueryTableName = segment.getAliasSegment().map(AliasSegment::getIdentifier).orElseGet(() -> new IdentifierValue(""));
68          tableBinderContexts.put(subqueryTableName.getValue().toLowerCase(),
69                  new SimpleTableSegmentBinderContext(createSubqueryProjections(boundedSelect.getProjections().getProjections(), subqueryTableName)));
70          return result;
71      }
72      
73      private static void fillPivotColumnNamesInBinderContext(final SubqueryTableSegment segment, final SQLStatementBinderContext statementBinderContext) {
74          segment.getPivot().ifPresent(optional -> optional.getPivotColumns().forEach(each -> statementBinderContext.getPivotColumnNames().add(each.getIdentifier().getValue().toLowerCase())));
75      }
76      
77      private static Collection<ProjectionSegment> createSubqueryProjections(final Collection<ProjectionSegment> projections, final IdentifierValue subqueryTableName) {
78          Collection<ProjectionSegment> result = new LinkedList<>();
79          for (ProjectionSegment each : projections) {
80              if (each instanceof ColumnProjectionSegment) {
81                  result.add(createColumnProjection((ColumnProjectionSegment) each, subqueryTableName));
82              } else if (each instanceof ShorthandProjectionSegment) {
83                  result.addAll(createSubqueryProjections(((ShorthandProjectionSegment) each).getActualProjectionSegments(), subqueryTableName));
84              } else {
85                  result.add(each);
86              }
87          }
88          return result;
89      }
90      
91      private static ColumnProjectionSegment createColumnProjection(final ColumnProjectionSegment originalColumn, final IdentifierValue subqueryTableName) {
92          ColumnSegment newColumnSegment = new ColumnSegment(0, 0, originalColumn.getAlias().orElseGet(() -> originalColumn.getColumn().getIdentifier()));
93          if (!Strings.isNullOrEmpty(subqueryTableName.getValue())) {
94              newColumnSegment.setOwner(new OwnerSegment(0, 0, subqueryTableName));
95          }
96          newColumnSegment.setColumnBoundedInfo(
97                  new ColumnSegmentBoundedInfo(originalColumn.getColumn().getColumnBoundedInfo().getOriginalDatabase(), originalColumn.getColumn().getColumnBoundedInfo().getOriginalSchema(),
98                          originalColumn.getColumn().getColumnBoundedInfo().getOriginalTable(), originalColumn.getColumn().getColumnBoundedInfo().getOriginalColumn()));
99          ColumnProjectionSegment result = new ColumnProjectionSegment(newColumnSegment);
100         result.setVisible(originalColumn.isVisible());
101         return result;
102     }
103 }
