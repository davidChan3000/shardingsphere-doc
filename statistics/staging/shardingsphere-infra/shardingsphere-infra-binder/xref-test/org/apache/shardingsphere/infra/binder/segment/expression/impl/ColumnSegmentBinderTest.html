44  class ColumnSegmentBinderTest {
45      
46      @Test
47      void assertBindWithMultiTablesJoinAndNoOwner() {
48          Map<String, TableSegmentBinderContext> tableBinderContexts = new LinkedHashMap<>();
49          ColumnSegment boundedOrderIdColumn = new ColumnSegment(0, 0, new IdentifierValue("order_id"));
50          boundedOrderIdColumn.setColumnBoundedInfo(new ColumnSegmentBoundedInfo(new IdentifierValue(DefaultDatabase.LOGIC_NAME), new IdentifierValue(DefaultDatabase.LOGIC_NAME),
51                  new IdentifierValue("t_order"), new IdentifierValue("order_id")));
52          tableBinderContexts.put("t_order", new SimpleTableSegmentBinderContext(Collections.singleton(new ColumnProjectionSegment(boundedOrderIdColumn))));
53          ColumnSegment boundedItemIdColumn = new ColumnSegment(0, 0, new IdentifierValue("item_id"));
54          boundedItemIdColumn.setColumnBoundedInfo(new ColumnSegmentBoundedInfo(new IdentifierValue(DefaultDatabase.LOGIC_NAME), new IdentifierValue(DefaultDatabase.LOGIC_NAME),
55                  new IdentifierValue("t_order_item"), new IdentifierValue("item_id")));
56          tableBinderContexts.put("t_order_item", new SimpleTableSegmentBinderContext(Collections.singleton(new ColumnProjectionSegment(boundedItemIdColumn))));
57          ColumnSegment columnSegment = new ColumnSegment(0, 0, new IdentifierValue("order_id"));
58          SQLStatementBinderContext statementBinderContext =
59                  new SQLStatementBinderContext(mock(ShardingSphereMetaData.class), DefaultDatabase.LOGIC_NAME, TypedSPILoader.getService(DatabaseType.class, "FIXTURE"), Collections.emptySet());
60          ColumnSegment actual = ColumnSegmentBinder.bind(columnSegment, SegmentType.JOIN_ON, statementBinderContext, tableBinderContexts, Collections.emptyMap());
61          assertNotNull(actual.getColumnBoundedInfo());
62          assertNull(actual.getOtherUsingColumnBoundedInfo());
63          assertThat(actual.getColumnBoundedInfo().getOriginalDatabase().getValue(), is(DefaultDatabase.LOGIC_NAME));
64          assertThat(actual.getColumnBoundedInfo().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));
65          assertThat(actual.getColumnBoundedInfo().getOriginalTable().getValue(), is("t_order"));
66          assertThat(actual.getColumnBoundedInfo().getOriginalColumn().getValue(), is("order_id"));
67      }
68      
69      @Test
70      void assertBindFromOuterTable() {
71          Map<String, TableSegmentBinderContext> outerTableBinderContexts = new LinkedHashMap<>();
72          ColumnSegment boundedOrderStatusColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
73          boundedOrderStatusColumn.setColumnBoundedInfo(new ColumnSegmentBoundedInfo(new IdentifierValue(DefaultDatabase.LOGIC_NAME), new IdentifierValue(DefaultDatabase.LOGIC_NAME),
74                  new IdentifierValue("t_order"), new IdentifierValue("status")));
75          outerTableBinderContexts.put("t_order", new SimpleTableSegmentBinderContext(Collections.singleton(new ColumnProjectionSegment(boundedOrderStatusColumn))));
76          ColumnSegment boundedOrderItemStatusColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
77          boundedOrderItemStatusColumn.setColumnBoundedInfo(new ColumnSegmentBoundedInfo(new IdentifierValue(DefaultDatabase.LOGIC_NAME), new IdentifierValue(DefaultDatabase.LOGIC_NAME),
78                  new IdentifierValue("t_order_item"), new IdentifierValue("status")));
79          outerTableBinderContexts.put("t_order_item", new SimpleTableSegmentBinderContext(Collections.singleton(new ColumnProjectionSegment(boundedOrderItemStatusColumn))));
80          SQLStatementBinderContext statementBinderContext =
81                  new SQLStatementBinderContext(mock(ShardingSphereMetaData.class), DefaultDatabase.LOGIC_NAME, TypedSPILoader.getService(DatabaseType.class, "FIXTURE"), Collections.emptySet());
82          ColumnSegment columnSegment = new ColumnSegment(0, 0, new IdentifierValue("status"));
83          ColumnSegment actual = ColumnSegmentBinder.bind(columnSegment, SegmentType.PROJECTION, statementBinderContext, Collections.emptyMap(), outerTableBinderContexts);
84          assertNotNull(actual.getColumnBoundedInfo());
85          assertNull(actual.getOtherUsingColumnBoundedInfo());
86          assertThat(actual.getColumnBoundedInfo().getOriginalDatabase().getValue(), is(DefaultDatabase.LOGIC_NAME));
87          assertThat(actual.getColumnBoundedInfo().getOriginalSchema().getValue(), is(DefaultDatabase.LOGIC_NAME));
88          assertThat(actual.getColumnBoundedInfo().getOriginalTable().getValue(), is("t_order_item"));
89          assertThat(actual.getColumnBoundedInfo().getOriginalColumn().getValue(), is("status"));
90      }
