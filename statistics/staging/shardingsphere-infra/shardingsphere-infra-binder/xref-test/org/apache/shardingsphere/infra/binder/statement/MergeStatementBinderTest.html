62      @Test
63      void assertBind() {
64          MergeStatement mergeStatement = new OracleMergeStatement();
65          SimpleTableSegment targetTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
66          targetTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("a")));
67          mergeStatement.setTarget(targetTable);
68          SimpleTableSegment sourceTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order_item")));
69          sourceTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("b")));
70          mergeStatement.setSource(sourceTable);
71          mergeStatement.setExpression(new ExpressionWithParamsSegment(0, 0, new BinaryOperationExpression(0, 0, new ColumnSegment(0, 0, new IdentifierValue("id")),
72                  new ColumnSegment(0, 0, new IdentifierValue("order_id")), "=", "id = order_id")));
73          UpdateStatement updateStatement = new OracleUpdateStatement();
74          updateStatement.setTable(targetTable);
75          ColumnSegment targetTableColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
76          targetTableColumn.setOwner(new OwnerSegment(0, 0, new IdentifierValue("a")));
77          ColumnSegment sourceTableColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
78          sourceTableColumn.setOwner(new OwnerSegment(0, 0, new IdentifierValue("b")));
79          SetAssignmentSegment setAssignmentSegment = new SetAssignmentSegment(0, 0,
80                  Collections.singletonList(new ColumnAssignmentSegment(0, 0, Collections.singletonList(targetTableColumn), sourceTableColumn)));
81          updateStatement.setSetAssignment(setAssignmentSegment);
82          updateStatement.setWhere(new WhereSegment(0, 0, new BinaryOperationExpression(0, 0, new ColumnSegment(0, 0, new IdentifierValue("item_id")),
83                  new LiteralExpressionSegment(0, 0, 1), "=", "item_id = 1")));
84          mergeStatement.setUpdate(updateStatement);
85          MergeStatement actual = new MergeStatementBinder().bind(mergeStatement, createMetaData(), DefaultDatabase.LOGIC_NAME);
86          assertThat(actual, not(mergeStatement));
87          assertThat(actual.getSource(), not(mergeStatement.getSource()));
88          assertThat(actual.getSource(), instanceOf(SimpleTableSegment.class));
89          assertThat(actual.getTarget(), not(mergeStatement.getTarget()));
90          assertThat(actual.getTarget(), instanceOf(SimpleTableSegment.class));
91          assertThat(actual.getUpdate(), not(mergeStatement.getUpdate()));
92          assertThat(actual.getUpdate().getSetAssignment().getAssignments().iterator().next().getValue(), instanceOf(ColumnSegment.class));
93          assertThat(((ColumnSegment) actual.getUpdate().getSetAssignment().getAssignments().iterator().next().getValue()).getColumnBoundedInfo().getOriginalTable().getValue(), is("t_order_item"));
94      }

117         MergeStatement mergeStatement = new OracleMergeStatement();
118         SimpleTableSegment targetTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
119         targetTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("a")));
120         mergeStatement.setTarget(targetTable);
121         ProjectionsSegment projectionsSegment = new ProjectionsSegment(0, 0);
122         ExpressionProjectionSegment expressionProjectionSegment = new ExpressionProjectionSegment(0, 0, "status + 1", new BinaryOperationExpression(0, 0,
123                 new ColumnSegment(0, 0, new IdentifierValue("status")), new LiteralExpressionSegment(0, 0, 1), "+", "status + 1"));
124         expressionProjectionSegment.setAlias(new AliasSegment(0, 0, new IdentifierValue("new_status")));
125         projectionsSegment.getProjections().add(expressionProjectionSegment);
126         OracleSelectStatement oracleSelectStatement = new OracleSelectStatement();
127         oracleSelectStatement.setProjections(projectionsSegment);
128         oracleSelectStatement.setFrom(new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order_item"))));
129         SubqueryTableSegment subqueryTableSegment = new SubqueryTableSegment(new SubquerySegment(0, 0, oracleSelectStatement, ""));
130         subqueryTableSegment.setAlias(new AliasSegment(0, 0, new IdentifierValue("b")));
131         mergeStatement.setSource(subqueryTableSegment);
132         UpdateStatement updateStatement = new OracleUpdateStatement();
133         ColumnSegment targetTableColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
134         targetTableColumn.setOwner(new OwnerSegment(0, 0, new IdentifierValue("a")));
135         ColumnSegment sourceTableColumn = new ColumnSegment(0, 0, new IdentifierValue("new_status"));
136         SetAssignmentSegment setAssignmentSegment = new SetAssignmentSegment(0, 0,
137                 Collections.singletonList(new ColumnAssignmentSegment(0, 0, Collections.singletonList(targetTableColumn), sourceTableColumn)));
138         updateStatement.setSetAssignment(setAssignmentSegment);
139         mergeStatement.setUpdate(updateStatement);
140         MergeStatement actual = new MergeStatementBinder().bind(mergeStatement, createMetaData(), DefaultDatabase.LOGIC_NAME);
141         assertThat(actual, not(mergeStatement));
142     }

144     @Test
145     void assertBindUpdateDeleteWhere() {
146         MergeStatement mergeStatement = new OracleMergeStatement();
147         SimpleTableSegment targetTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order")));
148         targetTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("a")));
149         mergeStatement.setTarget(targetTable);
150         SimpleTableSegment sourceTable = new SimpleTableSegment(new TableNameSegment(0, 0, new IdentifierValue("t_order_item")));
151         sourceTable.setAlias(new AliasSegment(0, 0, new IdentifierValue("b")));
152         mergeStatement.setSource(sourceTable);
153         OracleUpdateStatement updateStatement = new OracleUpdateStatement();
154         updateStatement.setTable(targetTable);
155         ColumnSegment targetTableColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
156         targetTableColumn.setOwner(new OwnerSegment(0, 0, new IdentifierValue("a")));
157         ColumnSegment sourceTableColumn = new ColumnSegment(0, 0, new IdentifierValue("status"));
158         sourceTableColumn.setOwner(new OwnerSegment(0, 0, new IdentifierValue("b")));
159         SetAssignmentSegment setAssignmentSegment = new SetAssignmentSegment(0, 0,
160                 Collections.singletonList(new ColumnAssignmentSegment(0, 0, Collections.singletonList(targetTableColumn), sourceTableColumn)));
161         updateStatement.setSetAssignment(setAssignmentSegment);
162         updateStatement.setDeleteWhere(new WhereSegment(0, 0, new BinaryOperationExpression(0, 0, new ColumnSegment(0, 0, new IdentifierValue("item_id")),
163                 new LiteralExpressionSegment(0, 0, 1), "=", "item_id = 1")));
164         mergeStatement.setUpdate(updateStatement);
165         MergeStatement actual = new MergeStatementBinder().bind(mergeStatement, createMetaData(), DefaultDatabase.LOGIC_NAME);
166         assertThat(actual.getUpdate(), instanceOf(OracleUpdateStatement.class));
167         assertThat(((OracleUpdateStatement) actual.getUpdate()).getDeleteWhere().getExpr(), instanceOf(BinaryOperationExpression.class));
168         assertThat(((BinaryOperationExpression) ((OracleUpdateStatement) actual.getUpdate()).getDeleteWhere().getExpr()).getLeft(), instanceOf(ColumnSegment.class));
169         assertThat(((ColumnSegment) ((BinaryOperationExpression) ((OracleUpdateStatement) actual.getUpdate()).getDeleteWhere().getExpr()).getLeft())
170                 .getColumnBoundedInfo().getOriginalTable().getValue(), is("t_order_item"));
171     }
